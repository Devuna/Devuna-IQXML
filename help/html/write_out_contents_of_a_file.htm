Write - Export to write TPS file contents

Here is an example of taking the contents of two files, and making an XML Document from it.

 

 

DOSTargetVariable        CSTRING(255)

 

! Have the user pick file name

! ============================

IF FILEDIALOG('Choose File to Export',DOSTargetVariable,'XML Files|*.XML',FILE:LongName + FILE:Save + FILE:KeepDir)

 

 IF instring('.XML,UPPER(DOSTargetVariable),1,1) = 0

         DOSTargetVariable = DOSTargetVariable  & '.XML'

 end

      

 if NOT XML:CreateXMLFile(DOSTargetVariable,'1.0','UTF-8')

 

         xml:CreateParent('IQSync');xml:AddParent()                ! Create a ROOT Tag for the application

 

 

         ! Write out file 1

         ! ================         

         xml:CreateParent('Files');xml:addParent()                        ! <Files>

         CLEAR(copydtl)

         SET(cpd:ID_KEY)

         LOOP

                 NEXT(COPYDTL)

                 IF ERRORCODE() THEN BREAK.

                 XML:AddFromGroup(CPD:RECORD,'File',True)                ! Write out each record <File>contents</File>

         end        

         xml:CloseParent()                                                ! </Files>

         

         ! Write out File 2

         ! ================         

         xml:CreateParent('Actions');xml:addParent()                ! <Actions>

         CLEAR(copyact)

         SET(CPA:Program_key)

         LOOP

                 NEXT(COPYACT)

                 IF ERRORCODE() THEN BREAK.

                 XML:AddFromGroup(CPA:RECORD,'Action',True)        ! Write out each record <Action>contents</Action>

         end

         xml:CloseParent()                                                ! </Actions>

         

         XML:CloseXMLFILE(True)                                        ! All Done - close </IQSync> tag too

         

         MESSAGE('The files are now exported.||File: ' & DOSTargetVariable,'Export Finished!',icon:asterisk)

 end   

END
